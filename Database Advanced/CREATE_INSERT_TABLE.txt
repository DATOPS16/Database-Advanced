CREATE TABLE Member (
    MemberID VARCHAR2(5) NOT NULL PRIMARY KEY,
    Name VARCHAR2(100) NOT NULL,
    MemberIC VARCHAR2(15) NOT NULL,
    MemberType VARCHAR2(10) NOT NULL CHECK (MemberType IN ('Standard','Premium')),
    PhoneNumber VARCHAR2(20) ,
    Address VARCHAR2(255),
    RegisterDate DATE NOT NULL,
    ExpiryDate DATE NOT NULL,
    Status VARCHAR2(20) NOT NULL CHECK (Status IN ('Active', 'Expired','Blacklisted')),
    BlackListDate DATE
);

CREATE TABLE Book (
    BookID VARCHAR2(5) NOT NULL PRIMARY KEY,
    Status VARCHAR2(20) NOT NULL CHECK (Status IN ('Available', 'Reserved')),
    BorrowedCount NUMBER,
    Remark VARCHAR2(255)
);

CREATE TABLE BookTitle (
    BookID VARCHAR2(5) NOT NULL PRIMARY KEY,
    Title VARCHAR2(255) NOT NULL,
    Author VARCHAR2(100) NOT NULL,
    PublishedYear NUMBER (4) NOT NULL,
    Category VARCHAR2(50) NOT NULL CHECK (Category IN ('Fiction', 'Comic', 'Mystery','Science','Fantasy')),  
    FOREIGN KEY (BookID) REFERENCES Book(BookID)
);

CREATE TABLE Librarian (
    LibrarianID VARCHAR2(5) NOT NULL PRIMARY KEY,
    Name VARCHAR2(100) NOT NULL,
    LibrarianIC VARCHAR2(15) NOT NULL,
    PhoneNumber VARCHAR2(20),
    Address VARCHAR2(255),
    Status VARCHAR(15) NOT NULL CHECK (Status IN ('Duty','Leave')),
    Salary NUMBER NOT NULL,
    LibrarianType VARCHAR2(15) NOT NULL CHECK (LibrarianType  IN ('FullTime','PartTime'))
);

CREATE TABLE BookReservation (
    BookReserveID VARCHAR2(5) NOT NULL PRIMARY KEY,
    MemberID VARCHAR2(5) NOT NULL,
    BookID VARCHAR2(5) NOT NULL,
    ReservationDate DATE DEFAULT SYSDATE NOT NULL,
    Status VARCHAR2(20) NOT NULL CHECK (Status IN ('Placed', 'Ready', 'Fulfilled','Cancelled')),
    FOREIGN KEY (MemberID) REFERENCES Member(MemberID),
    FOREIGN KEY (BookID) REFERENCES Book(BookID)
);

CREATE TABLE LibraryResource (
    ResourceID VARCHAR2(5) NOT NULL PRIMARY KEY,
    ResourceName VARCHAR2(100) NOT NULL UNIQUE,
    ResourceType VARCHAR2(20) NOT NULL CHECK (ResourceType IN ('Facility', 'Equipment')),
    Location VARCHAR2(255), -- Only applicable for facilities
    LibraryResourceStatus VARCHAR2(20) NOT NULL CHECK (LibraryResourceStatus IN ('Available', 'Under Maintenance', 'Closed'))
);

CREATE TABLE FacilityReservation (
    FacilityReserveID VARCHAR2(5) NOT NULL PRIMARY KEY,
    MemberID VARCHAR2(5) NOT NULL,
    ResourceID VARCHAR2(5) NOT NULL,
    ReservationDate DATE DEFAULT SYSDATE NOT NULL,
    ReservationStatus VARCHAR2(20) NOT NULL CHECK (ReservationStatus IN ('Ready','Cancelled')),
    FOREIGN KEY (MemberID) REFERENCES Member(MemberID),
    FOREIGN KEY (ResourceID) REFERENCES LibraryResource(ResourceID)
    );

CREATE TABLE Billing (
    BillingID VARCHAR2(5) NOT NULL PRIMARY KEY,
    MemberID VARCHAR2(5) NOT NULL,
    Type VARCHAR2(20) NOT NULL CHECK (Type IN ('overdue' , 'RenewMember', 'RegisterMember')),
    Amount NUMBER(10,2) NOT NULL,
    DueDate DATE NOT NULL,
    PayStatus VARCHAR2(20) NOT NULL CHECK (PayStatus  IN ('Paid', 'Unpaid',’Overdue’)),
    FOREIGN KEY (MemberID) REFERENCES Member(MemberID)
);

CREATE TABLE Invoice (
    InvoiceID VARCHAR2(5) NOT NULL PRIMARY KEY,
    PaymentDate DATE,
    PaymentMethod VARCHAR2(20) NOT NULL CHECK (PaymentMethod IN ('Cash' , 'Online Banking', 'e-wallet')),
    BillingID VARCHAR2(5) NOT NULL,
    FOREIGN KEY (BillingID) REFERENCES Billing(BillingID)
);

CREATE TABLE Technician (
    TechnicianID VARCHAR2(5) NOT NULL PRIMARY KEY,
    Name VARCHAR2(100) NOT NULL,
    Email VARCHAR2(100) UNIQUE,
    PhoneNumber VARCHAR2(20),
    Specialty VARCHAR2(30) NOT NULL CHECK (Specialty IN ('Equipment Repair','Facilities Administration'))
);

CREATE TABLE MaintenanceRequest (
    RequestID VARCHAR2(5) NOT NULL PRIMARY KEY,
    TechnicianID VARCHAR2(5) NOT NULL,
    ResourceID VARCHAR2(5) NOT NULL,
    LibrarianID VARCHAR2(5) NOT NULL,
    RequestDate DATE NOT NULL,
    Status VARCHAR2(20) NOT NULL CHECK (Status IN ('Pending', 'In Progress', 'Completed')),
    FOREIGN KEY (TechnicianID) REFERENCES Technician(TechnicianID),
    FOREIGN KEY (ResourceID) REFERENCES LibraryResource(ResourceID),
    FOREIGN KEY (LibrarianID) REFERENCES Librarian(LibrarianID)
);

CREATE TABLE BorrowingRecord (
    RecordID VARCHAR2(5) NOT NULL PRIMARY KEY,
    MemberID VARCHAR2(5) NOT NULL,
    LibrarianID VARCHAR2(5) NOT NULL,
    BookID VARCHAR2(5) NOT NULL,
    BorrowDate DATE NOT NULL,
    DueDate DATE NOT NULL,
    ReturnDate DATE,
    Status VARCHAR2(20) NOT NULL CHECK (Status IN ('Borrowed', 'Returned', 'Overdue')),
    ExtensionLoan VARCHAR2(3) NOT NULL CHECK (ExtensionLoan IN ('Yes', 'No')),
    FOREIGN KEY (MemberID) REFERENCES Member(MemberID),
    FOREIGN KEY (LibrarianID) REFERENCES Librarian(LibrarianID),
    FOREIGN KEY (BookID) REFERENCES Book(BookID)
);
